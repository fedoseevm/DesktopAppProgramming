            int b = int.Parse(textBox2.Text);

            if (CzyZaprzyjaznione(a, b))

        public bool CzyZaprzyjaznione(int a, int b)
        {
            return a == SumaDzielnikow(b) && b == SumaDzielnikow(a) && a > 0 && b > 0;
        }

        public int SumaDzielnikow(int x)
        {
            int suma = 0;
            for (int i = 1; i <= x / 2; i++)
            {
                if (x % i == 0)
                    suma += i;
            }
            return suma;
        }
